dyn/examples/lxmlsample.py: 

 1: from lxml import html
 2: 
 3: htmltext = """
 4: <html>
 5:     <head>
 6: 	<title>清水義孝の著書</title>
 7:     </head>
 8:     <body>
 9: 	<p class="title">
10: 	    <b>清水義孝の最新の著書には、次の本があります。</b>
11: 	</p>
12: 	<p class="recent books">
13: 	    <a class="book" href="https://www.amazon.co.jp/dp/B07TN4D3HG" id="link1">
14: 		Python3によるビジネスに役立つデータ分析入門
15: 	    </a>
16: 	    <a class="book" href="http://www.amazon.co.jp/dp/B07SRLRS4M" id="link2">
17: 		よくわかるPython3入門2.NumPy・Matplotlib編
18: 	    </a>
19: 	    <a class="book" href="http://www.amazon.co.jp/dp/B07T9SZ96B" id="link3">
20: 		よくわかるPython3入門4.Pandasでデータ分析編
21: 	    </a>
22: 	</p>
23: 	<p class="end">
24: 	    <b>そして、これらの本は好評発売中です。</b>
25: 	</p>
26:     </body>
27: </html>
28: """
29: 
30: load_html = html.fromstring(htmltext)
31: 
32: 
33: p_content = load_html.xpath("//p")
34: 
35: for content in p_content:
36:     print(content.attrib)
37: 
38: # {'class': 'title'}
39: # {'class': 'recent books'}
40: # {'class': 'end'}
41: 
42: 
43: 
44: ret=load_html.xpath("/html/head/title/text()")
45: print(ret)	
46: #['清水義孝の著書']
47: 
48:

50: ret=load_html.xpath('//a[@id="link1"]')
51: for a in ret:
